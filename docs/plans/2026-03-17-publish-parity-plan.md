```rust
/// Convert `![[image.png]]` and `![[image.png|300]]` to `<img>` tags.
/// Must run BEFORE resolve_transclusions since both match `![[...]]`.
fn convert_image_embeds(content: &str) -> String {
    transform_outside_fences(content, |line| {
        IMAGE_EMBED_RE
            .replace_all(line, |caps: &regex::Captures| {
                let filename = &caps[1];
                let size = caps.get(3).map(|m| m.as_str());

                let size_attrs = match size {
                    Some(s) if s.contains('x') => {
                        let parts: Vec<&str> = s.splitn(2, 'x').collect();
                        format!(r#" width="{}" height="{}""#, parts[0], parts[1])
                    }
                    Some(w) => format!(r#" width="{w}""#),
                    None => String::new(),
                };

                format!(r#"<img src="/assets/{filename}" alt=""{size_attrs} />"#)
            })
            .to_string()
    })
}
```

```rust
/// Search for an attachment file by walking up from the post's directory
/// and checking for an `attachment/` subdirectory.
fn find_attachment(start_dir: &Path, filename: &str) -> Option<std::path::PathBuf> {
    let mut dir = start_dir;
    loop {
        let candidate = dir.join("attachment").join(filename);
        if candidate.exists() {
            return Some(candidate);
        }
        // Also check the directory itself (image might be alongside notes)
        let candidate = dir.join(filename);
        if candidate.exists() {
            return Some(candidate);
        }
        dir = dir.parent()?;
    }
}
```

```rust
use std::sync::LazyLock;
use regex::Regex;

static HEADING_RE: LazyLock<Regex> =
    LazyLock::new(|| Regex::new(r"(?m)^#{1,6}\s+(.+)$").unwrap());

/// Slugify a heading the same way rehype-slug does:
/// lowercase, spaces to hyphens, strip non-alphanumeric (keep Korean + hyphens).
pub fn slugify_heading(heading: &str) -> String {
    heading
        .to_lowercase()
        .replace(' ', "-")
        .chars()
        .filter(|c| c.is_alphanumeric() || *c == '-' || is_korean(*c))
        .collect::<String>()
        .trim_matches('-')
        .to_string()
}

/// Extract all headings from a markdown post and return slugified heading IDs.
/// Handles duplicate headings by appending -1, -2, etc. (matching rehype-slug).
fn extract_headings(content: &str) -> Vec<String> {
    let mut slugs = Vec::new();
    let mut slug_counts: HashMap<String, usize> = HashMap::new();

    for cap in HEADING_RE.captures_iter(content) {
        let raw_heading = cap[1].trim();
        let base_slug = slugify_heading(raw_heading);

        let count = slug_counts.entry(base_slug.clone()).or_insert(0);
        let slug = if *count == 0 {
            base_slug.clone()
        } else {
            format!("{}-{}", base_slug, count)
        };
        *count += 1;
        slugs.push(slug);
    }

    slugs
}
```

```rust
    let mut blocks = HashMap::new();
```

```rust
static TRANSCLUSION_RE: LazyLock<Regex> =
    LazyLock::new(|| Regex::new(r"!\[\[([^\]#]+?)(?:#\^([a-zA-Z0-9-]+))?\]\]").unwrap());

fn resolve_transclusions(content: &str, index: &VaultIndex) -> String {
    transform_outside_fences(content, |line| {
        TRANSCLUSION_RE
            .replace_all(line, |caps: &regex::Captures| {
                let name = caps[1].trim();
                let block_id = caps.get(2).map(|m| m.as_str());

                // Check if this is an image embed (has file extension) — skip those
                let image_exts = ["png", "jpg", "jpeg", "gif", "svg", "webp"];
                if image_exts.iter().any(|ext| name.ends_with(&format!(".{ext}"))) {
                    return caps[0].to_string();
                }

                if let Some(&target_idx) = index.name_map.get(name) {
                    if let Some(bid) = block_id {
                        // Block-level transclusion — inline just that paragraph
                        if let Some(blocks) = index.block_map.get(name) {
                            if let Some(text) = blocks.get(bid) {
                                return text.clone();
                            }
                        }
                        // Block ID not found — fallback to plain text
                        format!("{name}")
                    } else {
                        // Full note transclusion
                        let target_content = &index.posts[target_idx].raw_content;
                        strip_frontmatter(target_content)
                    }
                } else {
                    format!("{name}")
                }
            })
            .to_string()
    })
}
```

/// Strip markdown/HTML for preview text.
fn strip_markdown_for_preview(content: &str) -> String {
    let body = crate::transform::strip_frontmatter(content);
    body.lines()
        .filter(|l| !l.starts_with('#') && !l.starts_with("```") && !l.starts_with("> "))
        .map(|l| {
            // Strip inline markdown
            l.replace("**", "")
                .replace("*", "")
                .replace("`", "")
                .replace("[[", "")
                .replace("]]", "")
                .replace("![[", "")
        })
        .collect::<Vec<_>>()
        .join(" ")
        .split_whitespace()
        .collect::<Vec<_>>()
        .join(" ")
}

/// Extract first sentence (up to first `.` or `。`, or 150 chars).
fn extract_first_sentence(text: &str) -> String {
    let max = 150;
    let end = text.char_indices()
        .find(|(i, c)| (*c == '.' || *c == '。') && *i > 10)
        .map(|(i, _)| i + 1)
        .unwrap_or_else(|| text.len().min(max));

    let result = &text[..end.min(text.len())];
    if result.len() >= max && end > max {
        format!("{}...", &text[..max])
    } else {
        result.to_string()
    }
}

```html
<script>
  let previewData: Record<string, { title: string; tags: string[]; summary: string }> | null = null;
  let tooltip: HTMLDivElement | null = null;
  let hoverTimer: ReturnType<typeof setTimeout> | null = null;

  async function ensurePreviewData() {
    if (previewData) return previewData;
    try {
      const res = await fetch("/previews.json");
      previewData = await res.json();
    } catch {
      previewData = {};
    }
    return previewData;
  }

  function createTooltip() {
    if (tooltip) return tooltip;
    tooltip = document.createElement("div");
    tooltip.className = "link-preview-tooltip";
    document.body.appendChild(tooltip);
    return tooltip;
  }

  function showPreview(anchor: HTMLAnchorElement) {
    const href = anchor.getAttribute("href") || "";
    const match = href.match(/^\/posts\/(.+?)(?:#|$)/);
    if (!match) return;
    const slug = match[1];

    hoverTimer = setTimeout(async () => {
      const data = await ensurePreviewData();
      if (!data) return;
      const preview = data[slug];
      if (!preview) return;

      const tip = createTooltip();
      // Clear previous content
      tip.replaceChildren();

      // Build tooltip content safely with DOM methods
      const titleEl = document.createElement("div");
      titleEl.className = "preview-title";
      titleEl.textContent = preview.title;
      tip.appendChild(titleEl);

      if (preview.tags.length) {
        const tagsEl = document.createElement("div");
        tagsEl.className = "preview-tags";
        for (const tag of preview.tags) {
          const tagSpan = document.createElement("span");
          tagSpan.textContent = tag;
          tagsEl.appendChild(tagSpan);
        }
        tip.appendChild(tagsEl);
      }

      const summaryEl = document.createElement("p");
      summaryEl.className = "preview-summary";
      summaryEl.textContent = preview.summary;
      tip.appendChild(summaryEl);

      const rect = anchor.getBoundingClientRect();
      const tipWidth = 280;
      let left = rect.left + window.scrollX;
      let top = rect.top + window.scrollY - 8;

      tip.style.left = `${Math.min(left, window.innerWidth - tipWidth - 16)}px`;
      tip.style.top = `${top}px`;
      tip.style.transform = "translateY(-100%)";
      tip.classList.add("visible");
    }, 300);
  }

  function hidePreview() {
    if (hoverTimer) {
      clearTimeout(hoverTimer);
      hoverTimer = null;
    }
    if (tooltip) {
      tooltip.classList.remove("visible");
    }
  }

  function initLinkPreviews() {
    document.querySelectorAll('a[href^="/posts/"]').forEach((a) => {
      a.addEventListener("mouseenter", () => showPreview(a as HTMLAnchorElement));
      a.addEventListener("mouseleave", hidePreview);
    });
  }

  initLinkPreviews();
  document.addEventListener("astro:after-swap", initLinkPreviews);
</script>
```

```rust
#[derive(Debug, Serialize)]
struct NavTreeNode {
    slug: String,
    title: String,
    is_hub: bool,
    children: Vec<NavTreeNode>,
}

#[derive(Debug, Serialize)]
struct NavTree {
    roots: Vec<NavTreeNode>,
    orphans: Vec<NavTreeNode>,
}

fn build_nav_tree(index: &VaultIndex, graph: &LinkGraph) -> NavTree {
    // Find all hubs
    let hub_indices: Vec<usize> = index.posts.iter().enumerate()
        .filter(|(_, p)| p.is_hub)
        .map(|(i, _)| i)
        .collect();

    // Build hub children: posts whose hub_parent matches a hub title,
    // OR posts that appear in a hub's forward_links
    let mut claimed: std::collections::HashSet<usize> = std::collections::HashSet::new();

    fn build_node(
        post_idx: usize,
        index: &VaultIndex,
        graph: &LinkGraph,
        claimed: &mut std::collections::HashSet<usize>,
    ) -> NavTreeNode {
        claimed.insert(post_idx);
        let post = &index.posts[post_idx];
        let mut children = Vec::new();

        if post.is_hub {
            // Children are posts linked from this hub
            for link in &graph.forward_links[post_idx] {
                if let Some(&child_idx) = index.slug_map.get(&link.target_slug) {
                    if !claimed.contains(&child_idx) {
                        children.push(build_node(child_idx, index, graph, claimed));
                    }
                }
            }

            // Also include posts that declare this hub as parent
            for (i, p) in index.posts.iter().enumerate() {
                if p.hub_parent.as_deref() == Some(&post.title) && !claimed.contains(&i) {
                    children.push(build_node(i, index, graph, claimed));
                }
            }
        }

        NavTreeNode {
            slug: post.slug.clone(),
            title: post.title.clone(),
            is_hub: post.is_hub,
            children,
        }
    }

    // Build roots (top-level hubs with no hub_parent)
    let roots: Vec<NavTreeNode> = hub_indices.iter()
        .filter(|&&i| index.posts[i].hub_parent.is_none())
        .map(|&i| build_node(i, index, graph, &mut claimed))
        .collect();

    // Orphans: posts not claimed by any hub
    let orphans: Vec<NavTreeNode> = index.posts.iter().enumerate()
        .filter(|(i, _)| !claimed.contains(i))
        .map(|(_, p)| NavTreeNode {
            slug: p.slug.clone(),
            title: p.title.clone(),
            is_hub: false,
            children: Vec::new(),
        })
        .collect();

    NavTree { roots, orphans }
}
```

```tsx
import { useState, useEffect } from "preact/hooks";
import type { NavTreeData, NavTreeNode } from "../lib/types";

interface Props {
  currentSlug: string;
}

function TreeNode({ node, currentSlug, defaultOpen }: {
  node: NavTreeNode;
  currentSlug: string;
  defaultOpen: boolean;
}) {
  const [open, setOpen] = useState(defaultOpen);
  const isCurrent = node.slug === currentSlug;
  const hasChildren = node.children.length > 0;

  return (
    <li class="nav-tree-item">
      <div class={`nav-tree-label ${isCurrent ? "current" : ""}`}>
        {hasChildren && (
          <button
            class="nav-tree-toggle"
            onClick={() => setOpen(!open)}
            aria-label={open ? "Collapse" : "Expand"}
          >
            <svg width="12" height="12" viewBox="0 0 24 24" fill="none"
              stroke="currentColor" stroke-width="2" style={{
                transform: open ? "rotate(90deg)" : "rotate(0deg)",
                transition: "transform 0.15s ease"
              }}>
              <polyline points="9 18 15 12 9 6" />
            </svg>
          </button>
        )}
        {!hasChildren && <span class="nav-tree-spacer" />}
        <a href={`/${node.is_hub ? "hubs" : "posts"}/${node.slug}`}>
          {node.title}
        </a>
      </div>
      {hasChildren && open && (
        <ul class="nav-tree-children">
          {node.children.map((child) => (
            <TreeNode
              key={child.slug}
              node={child}
              currentSlug={currentSlug}
              defaultOpen={isAncestor(child, currentSlug)}
            />
          ))}
        </ul>
      )}
    </li>
  );
}

/** Check if a node or any descendant matches the target slug. */
function isAncestor(node: NavTreeNode, targetSlug: string): boolean {
  if (node.slug === targetSlug) return true;
  return node.children.some((c) => isAncestor(c, targetSlug));
}

export default function NavTree({ currentSlug }: Props) {
  const [tree, setTree] = useState<NavTreeData | null>(null);

  useEffect(() => {
    fetch("/nav-tree.json")
      .then((r) => r.json())
      .then((data: NavTreeData) => setTree(data))
      .catch(() => {});
  }, []);

  if (!tree) return null;
  if (tree.roots.length === 0) return null;

  return (
    <nav class="nav-tree">
      <h3 class="nav-tree-heading">탐색</h3>
      <ul class="nav-tree-root">
        {tree.roots.map((root) => (
          <TreeNode
            key={root.slug}
            node={root}
            currentSlug={currentSlug}
            defaultOpen={isAncestor(root, currentSlug)}
          />
        ))}
      </ul>
    </nav>
  );
}
```
